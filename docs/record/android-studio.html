public static String getHtml(String path) throws Exception { HttpURLConnection connection = null; BufferedReader reader = null; URL url = new URL(path); connection = (HttpURLConnection) url.openConnection(); //设置请求方法 connection.setRequestMethod("GET"); //设置连接超时时间（毫秒） connection.setConnectTimeout(5000); //设置读取超时时间（毫秒） connection.setReadTimeout(5000); //返回输入流 InputStream in; if(connection.getResponseCode()==HttpURLConnection.HTTP_OK){ in = connection.getInputStream(); }else{//读取错误流 错误了，但服务器还是返回了数据 in = connection.getErrorStream(); } //读取输入流 reader = new BufferedReader(new InputStreamReader(in)); StringBuilder result = new StringBuilder(); String line; while ((line = reader.readLine()) != null) { result.append(line); } return (result.toString()); }
