requireTs(['./机器人.ts'], function ({default:{
    runRobot,
    VillageState,
    goalOrientedRobot

        count += runRobot(VillageState.random(),goalOrientedRobot, [])
    }

    console.log(`万次平均步数${count / 10000}`);
});
